How It Works

Define semantic bundles. Get indexed, LLM-ready output.

srcpack.config.ts
import { defineConfig } from "srcpack";

export default defineConfig({
  bundles: {
    web: ["app/**", "components/**"],
    api: ["server/**", "routes/**"],
    db: ["prisma/**", "migrations/**"],
  },
});

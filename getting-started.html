Getting Started ​

Quick Start ​

ts
// srcpack.config.ts
import { defineConfig } from "srcpack";

export default defineConfig({
  bundles: {
    web: ["src/client/**/*", "src/shared/**/*"],
    api: ["src/server/**/*", "src/shared/**/*"],
    db: ["src/db/**/*", "prisma/schema.prisma"],
  },

  // Optional: auto-upload to Google Drive
  upload: {
    gdrive: { folder: "1abc..." },
  },
});
sh
npx srcpack          # Bundle all, upload if configured
npx srcpack web api  # Bundle specific domains only
